CREATE TABLE Seeker (
`S_No` int(11) AUTO_INCREMENT PRIMARY KEY,
`Seeker_ID` varchar(25) NOT NULL UNIQUE,
`Reg_Time` timestamp NOT NULL DEFAULT current_timestamp(),
`First_Name` varchar(50) NOT NULL,
`Last_Name` varchar(50) NOT NULL,
`Date_Of_Birth` date NOT NULL,
`Age` int(11) NOT NULL,
`Gender` varchar(10) NOT NULL,
`Phone_Number` varchar(15) NOT NULL,
`E_Mail` varchar(100) NOT NULL,
`Password` varchar(225) NOT NULL,
`Profile_Pic` varchar(100) NOT NULL,
`City` varchar(50) NOT NULL,
`Country` varchar(50) NOT NULL,
`Seeker_Type` ENUM('School Student','College Student','Working Professional') NOT NULL,
`Skill` text NOT NULL,
`Certification` varchar(150) NOT NULL,
`Resume` varchar(150) DEFAULT NULL,
`Access_Gate` varchar(50) NOT NULL DEFAULT 'Grant',
`Verification_Code` varchar(50) DEFAULT NULL,
`Login_Status` varchar(20) DEFAULT NULL,
`Checking_Status` varchar(10) DEFAULT NULL);
